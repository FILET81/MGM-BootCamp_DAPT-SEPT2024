def roc_auc_curve(y_pred_prob, title, threshold = 0.5):
    # Convert probabilities predictions to binary (0 or 1) based on a threshold
    y_pred_class = (y_pred_prob > threshold).astype(int)
 
    # Calculate ROC curve
    fpr, tpr, thresholds = roc_curve(y_test, y_pred_prob)
    roc_auc = auc(fpr, tpr)
 
    # Plot ROC curve
    plt.figure(figsize=(10, 7))
    plt.plot(fpr, tpr, color='darkorange', lw=2, label='ROC curve (area = {:.2f})'.format(roc_auc))
    plt.plot([0, 1], [0, 1], color='navy', lw=2, linestyle='--')
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title(title)
    plt.legend(loc="lower right")
    plt.show()

# Set the max depth for the rf
n_estimators = 25

rf = RandomForestClassifier(n_estimators=n_estimators, random_state=0, class_weight='balanced')

# Fit rf to the training set    
rf.fit(X_train_std, y_train) 

# Get feature importances
feature_importances = rf.feature_importances_

# Create a dictionary to associate feature names with importances
feature_importance_dict = dict(zip(feature_names, feature_importances))

# Round feature importances to 3 decimal places
rounded_importances = {feature: round(importance, 3) for feature, importance in feature_importance_dict.items()}

# Get features with importance greater than 0
positive_importances = {feature: importance for feature, importance in rounded_importances.items() if importance > 0}

# Sort positive importances
sorted_positive_importances = sorted(positive_importances.items(), key=lambda x: x[1], reverse=False)

# Plot the positive feature importances
plt.figure(figsize=(10, 10))
plt.barh(range(len(sorted_positive_importances)), [importance for _, importance in sorted_positive_importances], align='center')
plt.yticks(range(len(sorted_positive_importances)), [feature for feature, _ in sorted_positive_importances])
plt.xlabel('Feature Importance')
plt.title('Features with Importance > 0')
plt.show()

We see that there are many more feature importances in comparison to the tree model. This will be further explored in the main notebook. # Define the hyperparameter grid

# Number of trees in random forest

max_depth = [5, 10, 20]

random_grid = {'n_estimators': n_estimators,
               'max_depth': max_depth}

# Create a DecisionTreeClassifier
rf = RandomForestClassifier(random_state=0, class_weight='balanced')

# Create a RandomizedSearchCV object
random_search = RandomizedSearchCV(rf, param_distributions=random_grid, n_iter=8, verbose=3, cv=4, random_state=0)

# Fit the RandomizedSearchCV object to the data
